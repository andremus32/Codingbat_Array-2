//Array-2 > evenOdd
public int[] evenOdd(int[] nums) {
  int countOdd = 0;
  int countEven = 0;
  int[] arr = new int[nums.length];
  for(int i = 0; i < nums.length; i++){
    if(nums[i] % 2 == 0){
      arr[countEven] = nums[i];
      countEven++;
    }
    else {
      arr[nums.length - 1 - countOdd] = nums[i];
      countOdd++;
    }
  }
  return arr;
}
