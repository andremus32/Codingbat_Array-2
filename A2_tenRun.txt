//Array-2 > tenRun
public int[] tenRun(int[] nums) {
  int tmp = 0;
  boolean isFirstTenMet = false;
  for(int i = 0; i < nums.length; i++){
    if(!isFirstTenMet && nums[i] % 10 == 0){
      isFirstTenMet = true;
      tmp = nums[i];
    }
    if(isFirstTenMet && nums[i] % 10 == 0){
      tmp = nums[i];
    }
    else if(isFirstTenMet){
      nums[i] = tmp;
    }
  }
  return nums;
}
