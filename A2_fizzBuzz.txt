//Array-2 > fizzBuzz
public String[] fizzBuzz(int start, int end) {
  String[] arr = new String[end - start];
  for(int i = 0; i < end-start; i++){
    if((i+start) % 3 == 0 && (i+start) % 5 == 0){
      arr[i] = "FizzBuzz";
    }
    else if((i+start) % 3 == 0){
      arr[i] = "Fizz";
    }
    else if((i+start) % 5 == 0){
      arr[i] = "Buzz";
    }
    else arr[i] = String.valueOf(i+start);
  }
  return arr;
}
