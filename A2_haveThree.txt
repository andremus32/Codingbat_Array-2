//Array-2 > haveThree
public boolean haveThree(int[] nums) {
  boolean isThreeNext = true;
  int count = 0;
  for(int i = 0; i < nums.length; i++){
      if(i < nums.length - 1 && nums[i] == 3 && nums[i+1] == 3){
        isThreeNext = false;
        break;
      }
    else if(nums[i] == 3){
      count++;
    }
  }
  return (isThreeNext && count == 3);
}
