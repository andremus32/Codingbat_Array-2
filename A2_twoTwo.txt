//Array-2 > twoTwo
public boolean twoTwo(int[] nums) {
  boolean check = true;
  if(nums.length > 1){
    if((nums[0] == 2 && nums[1] != 2) || (nums[nums.length-1] == 2 && nums[nums.length-2] != 2)){
      check = false;
    }
    else{
      for(int i = 1; i < nums.length-1; i++){
        if(nums[i] == 2 && nums[i-1] != 2 && nums[i+1] != 2){
          check = false;
          break;
        }
      }
    }
  }
  else if(nums.length == 1 && nums[0] == 2){
    check = false;
  }
return check;
}
